✅ 1. FCFS (First Come First Serve)
#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head;
    int requests[100];
    int i, total_seek = 0;
    float avg_seek;

    // Input
    printf("Enter total number of requests: ");
    scanf("%d", &n);

    printf("Enter request sequence:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &requests[i]);
    }

    printf("Enter initial head position: ");
    scanf("%d", &head);

    // Output Table
    printf("\n%-20s %-25s\n", "Next Track", "Tracks Traversed");
    printf("-------------------------------------------------\n");

    int current = head, distance;
    printf("%-20d %-25d\n", current, 0);

    for(i = 0; i < n; i++) {
        distance = abs(current - requests[i]);
        total_seek += distance;
        printf("%-20d %-25d\n", requests[i], distance);
        current = requests[i];
    }

    // Results
    avg_seek = (float) total_seek / n;

    printf("\nTotal Seek Length = %d\n", total_seek);
    printf("Avg Seek Len = %.2f\n", avg_seek);

    return 0;
}

✅ 2. SSTF (Shortest Seek Time First)
#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head;
    int requests[100];
    int visited[100] = {0};
    int i, total_seek = 0;
    float avg_seek;

    printf("total number of requests: ");
    scanf("%d", &n);

    printf("request sequence:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &requests[i]);
    }

    printf("initial head position: ");
    scanf("%d", &head);

    printf("\n%-20s %-25s\n", "Next Track", "Tracks Traversed");
    printf("-------------------------------------------------\n");

    int current = head, distance, min_dist, index;
    printf("%-20d %-25d\n", current, 0);

    for(i = 0; i < n; i++) {
        min_dist = 10000;

        for(int j = 0; j < n; j++) {
            if(!visited[j]) {
                distance = abs(current - requests[j]);
                if(distance < min_dist) {
                    min_dist = distance;
                    index = j;
                }
            }
        }

        visited[index] = 1;
        total_seek += min_dist;

        printf("%-20d %-25d\n", requests[index], min_dist);
        current = requests[index];
    }

    avg_seek = (float) total_seek / n;

    printf("\nTotal Seek Length = %d\n", total_seek);
    printf("Avg Seek Len = %.2f\n", avg_seek);

    return 0;
}

✅ 3. SCAN (Elevator Algorithm)
#include <stdio.h>
#include <stdlib.h>

void sort(int arr[], int n) {
    int i, j, temp;
    for(i = 0; i < n - 1; i++) {
        for(j = 0; j < n - i - 1; j++) {
            if(arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main() {
    int n, head, disk_size;
    int requests[100];
    int left[100], right[100];
    int l = 0, r = 0;
    int i, total_seek = 0;

    printf("Enter total number of requests: ");

        scanf("%d", &requests[i]);
    }

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter disk size: ");
    scanf("%d", &disk_size);

    sort(requests, n);

    for(i = 0; i < n; i++) {
        if(requests[i] < head)
            left[l++] = requests[i];
        else
            right[r++] = requests[i];
    }

    printf("\n%-20s %-25s\n", "Next Track", "Tracks Traversed");
    printf("-------------------------------------------------\n");

    int current = head, distance;

    distance = abs(current - (disk_size - 1));

    return 0;
}

✅ 6. CSCAN
#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, i, j;
    int request[100];
    int head;
    int totalSeekLength = 0, movement, prev;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter the request sequence:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &request[i]);
    }

    printf("Enter initial head position: ");
    scanf("%d", &head);

    for(i = 0; i < n - 1; i++) {
        for(j = 0; j < n - i - 1; j++) {
            if(request[j] > request[j + 1]) {
                int temp = request[j];
                request[j] = request[j + 1];
                request[j + 1] = temp;
            }
        }
    }

    printf("\n%-20s %-25s\n", "Current Head", "Head Movement");
    printf("%-20d %-25s\n", head, "-");

    int pos = n;
    for(i = 0; i < n; i++) {
        if(request[i] >= head) {
            pos = i;
            break;
        }
    }

    for(i = pos; i < n; i++) {
        prev = head;
        head = request[i];
        movement = abs(head - prev);
        printf("%-20d %-25d\n", head, movement);
        totalSeekLength += movement;
    }

    if(pos > 0) {
        prev = head;
        head = request[0];
        movement = abs(head - prev);
        printf("%-20d %-25d\n", head, movement);
        totalSeekLength += movement;
    }

    for(i = 1; i < pos; i++) {
        prev = head;
        head = request[i];
        movement = abs(head - prev);
        printf("%-20d %-25d\n", head, movement);
        totalSeekLength += movement;
    }

    printf("\nTotal Seek Length = %d\n", totalSeekLength);
    printf("Average Seek Length = %.2f\n", (float) totalSeekLength / n);

    return 0;
}
