✅ FIFO Page Replacement
#include <stdio.h>

int main() {
    int n, f;

    printf("Enter number of Pages: ");
    scanf("%d", &n);

    int pages[n];
    printf("Enter Pages:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &pages[i]);
    }

    printf("Enter number of frames: ");
    scanf("%d", &f);

    int frame[f];
    int index = 0, faults = 0, hits = 0;

    // Initialize frames
    for (int i = 0; i < f; i++) {
        frame[i] = -1;
    }

    printf("Page\tFrame\t\t\tStatus\n");

    // Processing
    for (int i = 0; i < n; i++) {
        int page = pages[i];
        int found = 0;

        // Check Hit
        for (int j = 0; j < f; j++) {
            if (frame[j] == page) {
                found = 1;
                hits++;
                break;
            }
        }

        // If Fault
        if (!found) {
            frame[index] = page;
            index = (index + 1) % f;
            faults++;
        }

        // Output
        printf("%d\t", page);
        for (int j = 0; j < f; j++) {
            if (frame[j] != -1)
                printf("%d ", frame[j]);
            else
                printf("- ");
        }

        if (found)
            printf("\t\tHIT\n");
        else
            printf("\t\tFAULT\n");
    }

    printf("Total Page Faults = %d\n", faults);
    printf("Total Page Hits = %d\n", hits);
    printf("Hit Ratio = %.2f\n", (float)hits / n);
    printf("Fault Ratio = %.2f\n", (float)faults / n);

    return 0;
}

✅ LRU Page Replacement
#include <stdio.h>

int main() {
    int n, f;

    printf("Enter number of pages: ");
    scanf("%d", &n);

    int pages[n];
    printf("Enter pages:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &pages[i]);
    }

    printf("Enter number of frames: ");
    scanf("%d", &f);

    int frame[f], time[f];
    int count = 0, faults = 0, hits = 0;

    // Initialize
    for (int i = 0; i < f; i++) {
        frame[i] = -1;
        time[i] = 0;
    }

    printf("Page\tFrames\t\t\tStatus\n");

    // Processing
    for (int i = 0; i < n; i++) {
        int page = pages[i];
        int found = 0;

        // Check HIT
        for (int j = 0; j < f; j++) {
            if (frame[j] == page) {
                count++;
                time[j] = count;
                found = 1;
                hits++;
                break;
            }
        }

        // If FAULT
        if (!found) {
            int pos = -1;

            // Check empty frame
            for (int j = 0; j < f; j++) {
                if (frame[j] == -1) {
                    pos = j;
                    break;
                }
            }

            // Replace LRU
            if (pos == -1) {
                int min = time[0];
                pos = 0;

                for (int j = 1; j < f; j++) {
                    if (time[j] < min) {
                        min = time[j];
                        pos = j;
                    }
                }
            }

            frame[pos] = page;
            count++;
            time[pos] = count;
            faults++;
        }

        // Output
        printf("%d\t", page);
        for (int j = 0; j < f; j++) {
            if (frame[j] != -1)
                printf("%d ", frame[j]);
            else
                printf("- ");
        }

        if (found)
            printf("\t\tHIT\n");
        else
            printf("\t\tFAULT\n");
    }

    printf("Total Page Faults = %d\n", faults);
    printf("Total Page Hits = %d\n", hits);
    printf("Hit Ratio = %.2f\n", (float)hits / n);
    printf("Fault Ratio = %.2f\n", (float)faults / n);

    return 0;
}

#include <stdio.h>

int main() {
    int n, f;

    printf("Enter number of Pages: ");
    scanf("%d", &n);

    int pages[n];
    printf("Enter Pages:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &pages[i]);
    }

    printf("Enter the Number of Frames: ");
    scanf("%d", &f);

    int frame[f];
    int faults = 0, hits = 0;

    // Initialize
    for (int i = 0; i < f; i++) {
        frame[i] = -1;
    }

    printf("Page\tFrames\t\t\tStatus\n");

    // Processing
    for (int i = 0; i < n; i++) {
        int page = pages[i];
        int found = 0;

        // Check Hit
        for (int j = 0; j < f; j++) {
            if (frame[j] == page) {
                found = 1;
                hits++;
                break;
            }
        }

        // If Fault
        if (!found) {
            int pos = -1;

            // Check empty
            for (int j = 0; j < f; j++) {
                if (frame[j] == -1) {
                    pos = j;
                    break;
                }
            }

            // Optimal logic
            if (pos == -1) {
                int farthest = -1;
                pos = 0;

                for (int j = 0; j < f; j++) {
                    int k;
                    for (k = i + 1; k < n; k++) {
                        if (frame[j] == pages[k]) {
                            if (k > farthest) {
                                farthest = k;
                                pos = j;
                            }
                            break;
                        }
                    }

                    if (k == n) {
                        pos = j;
                        break;
                    }
                }
            }

            frame[pos] = page;
            faults++;
        }

        // Output
        printf("%d\t", page);
        for (int j = 0; j < f; j++) {
            if (frame[j] != -1)
                printf("%d ", frame[j]);
            else
                printf("- ");
        }

        if (found)
            printf("\t\tHIT\n");
        else
            printf("\t\tFAULT\n");
    }

    printf("Total Page Faults = %d\n", faults);
    printf("Total Page Hits = %d\n", hits);
    printf("Hit Ratio = %.2f\n", (float)hits / n);
    printf("Fault Ratio = %.2f\n", (float)faults / n);

    return 0;
}
