✅ FCFS (First Come First Serve)
#include <stdio.h>

int main() {
    int n, i, time = 0, completed = 0;

    // INPUT
    printf("Enter number of processes: ");
    scanf("%d", &n);

    int at[n], bt[n], ct[n], tat[n], wt[n], vis[n];

    for (i = 0; i < n; i++) {
        printf("Enter AT and BT for P%d: ", i + 1);
        scanf("%d %d", &at[i], &bt[i]);
        vis[i] = 0;
    }

    // INITIALIZATION
    int gp[100], gt[100], k = 0;
    gt[0] = 0;
    float totalWT = 0, totalTAT = 0;

    // SCHEDULING
    while (completed < n) {
        int idx = -1;

        for (i = 0; i < n; i++) {
            if (at[i] <= time && vis[i] == 0) {
                if (idx == -1 || at[i] < at[idx] || 
                   (at[i] == at[idx] && i < idx)) {
                    idx = i;
                }
            }
        }

        if (idx == -1) {
            gp[k] = -1;
            time++;
            gt[k + 1] = time;
            k++;
            continue;
        }

        gp[k] = idx + 1;
        time += bt[idx];
        gt[k + 1] = time;
        k++;

        ct[idx] = time;
        tat[idx] = ct[idx] - at[idx];
        wt[idx] = tat[idx] - bt[idx];

        totalWT += wt[idx];
        totalTAT += tat[idx];

        vis[idx] = 1;
        completed++;
    }

    // OUTPUT
    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");
    for (i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
               i + 1, at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nGantt Chart:\n");
    for (i = 0; i < k; i++) {
        if (gp[i] == -1) printf("|  -  ");
        else printf("| P%-2d ", gp[i]);
    }
    printf("|\n");

    printf("%d", gt[0]);
    for (i = 1; i <= k; i++) {
        printf("     %d", gt[i]);
    }

    printf("\nAverage WT = %.2f", totalWT / n);
    printf("\nAverage TAT = %.2f\n", totalTAT / n);

    return 0;
}

✅ PP (Priority Preemptive)
#include <stdio.h>

int main() {
    int n, i, time = 0, completed = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    int at[n], bt[n], pr[n], rt[n], ct[n], tat[n], wt[n];

    for (i = 0; i < n; i++) {
        printf("Enter AT, BT, Priority for P%d: ", i + 1);
        scanf("%d %d %d", &at[i], &bt[i], &pr[i]);
        rt[i] = bt[i];
    }

    float totalWT = 0, totalTAT = 0;

    while (completed < n) {
        int best = 9999, idx = -1;

        for (i = 0; i < n; i++) {
            if (at[i] <= time && rt[i] > 0 && pr[i] < best) {
                best = pr[i];
                idx = i;
            }
        }

        if (idx == -1) {
            time++;
            continue;
        }

        time++;
        rt[idx]--;

        if (rt[idx] == 0) {
            ct[idx] = time;
            tat[idx] = ct[idx] - at[idx];
            wt[idx] = tat[idx] - bt[idx];

            totalWT += wt[idx];
            totalTAT += tat[idx];

            completed++;
        }
    }

    // OUTPUT
    printf("\nP\tAT\tBT\tPR\tCT\tTAT\tWT\n");
    for (i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t%d\t%d\t%d\t%d\n",
               i + 1, at[i], bt[i], pr[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage WT = %.2f", totalWT / n);
    printf("\nAverage TAT = %.2f\n", totalTAT / n);

    return 0;
}

✅ SRTF (Shortest Remaining Time First - Preemptive)
#include <stdio.h>

int main() {
    int n, i, time = 0, completed = 0;

    // INPUT
    printf("Enter number of processes: ");
    scanf("%d", &n);

    int at[n], bt[n], rt[n], ct[n], tat[n], wt[n];

    for (i = 0; i < n; i++) {
        printf("Enter AT and BT for P%d: ", i + 1);
        scanf("%d %d", &at[i], &bt[i]);
        rt[i] = bt[i];
    }

    // INITIALIZATION
    int gp[200], gt[200], k = 0;
    gt[0] = 0;

    float totalWT = 0, totalTAT = 0;

    // SCHEDULING
    while (completed < n) {
        int min = 9999, idx = -1;

        // PROCESS SELECTION
        for (i = 0; i < n; i++) {
            if (at[i] <= time && rt[i] > 0) {
                if (idx == -1 || rt[i] < min ||
                   (rt[i] == min && at[i] < at[idx]) ||
                   (rt[i] == min && at[i] == at[idx] && i < idx)) {
                    min = rt[i];
                    idx = i;
                }
            }
        }

        // CPU IDLE
        if (idx == -1) {
            gp[k] = -1;
            time++;
            gt[k + 1] = time;
            k++;
            continue;
        }

        // EXECUTION
        gp[k] = idx + 1;
        time++;
        rt[idx]--;
        gt[k + 1] = time;
        k++;

        // COMPLETION
        if (rt[idx] == 0) {
            ct[idx] = time;
            tat[idx] = ct[idx] - at[idx];
            wt[idx] = tat[idx] - bt[idx];

            totalWT += wt[idx];
            totalTAT += tat[idx];
            completed++;
        }
    }

    // OUTPUT
    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");
    for (i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
               i + 1, at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nGantt Chart:\n");
    for (i = 0; i < k; i++) {
        if (gp[i] == -1) printf("|  -  ");
        else printf("| P%-2d ", gp[i]);

    printf("|\n");

    printf("%d", gt[0]);
    for (i = 1; i <= k; i++) {
        printf("     %d", gt[i]);
    }

    printf("\nAverage WT = %.2f", totalWT / n);
    printf("\nAverage TAT = %.2f\n", totalTAT / n);

    return 0;

✅ RR (Round Robin)
#include <stdio.h>

int main() {
    int n, i, time = 0, completed = 0, tq;

    // INPUT
    printf("Enter number of processes: ");
    scanf("%d", &n);

    int at[n], bt[n], rt[n], ct[n], tat[n], wt[n];

    for (i = 0; i < n; i++) {
        printf("AT and BT for P%d: ", i + 1);
        scanf("%d %d", &at[i], &bt[i]);
        rt[i] = bt[i];
    }

    printf("Enter Time Quantum: ");
    scanf("%d", &tq);

    // INITIALIZATION
    int q[100], front = 0, rear = 0;
    int inQueue[n];

    for (i = 0; i < n; i++)
        inQueue[i] = 0;

    int gp[200], gt[200], k = 0;
    gt[0] = 0;

    float totalWT = 0, totalTAT = 0;

    // INITIAL QUEUE
    for (i = 0; i < n; i++) {
        if (at[i] == 0) {
            q[rear++] = i;
            inQueue[i] = 1;
        }
    }

    // SCHEDULING
    while (completed < n) {

        // CPU IDLE
        if (front == rear) {
            gp[k] = -1;
            time++;
            gt[k + 1] = time;
            k++;

            for (i = 0; i < n; i++) {
                if (at[i] <= time && rt[i] > 0 && !inQueue[i]) {
                    q[rear++] = i;
                    inQueue[i] = 1;
                }
            }
            continue;
        }

        // EXECUTION
        int idx = q[front++];
        int exec = (rt[idx] > tq) ? tq : rt[idx];

        for (int j = 0; j < exec; j++) {
            gp[k] = idx + 1;
            time++;
            rt[idx]--;

            gt[k + 1] = time;
            k++;

            for (i = 0; i < n; i++) {
                if (at[i] == time && rt[i] > 0 && !inQueue[i]) {
                    q[rear++] = i;
                    inQueue[i] = 1;
                }
            }
        }

        // COMPLETION / REQUEUE
        if (rt[idx] > 0) {
            q[rear++] = idx;
        } else {
            ct[idx] = time;
            tat[idx] = ct[idx] - at[idx];
            wt[idx] = tat[idx] - bt[idx];

            totalWT += wt[idx];
            totalTAT += tat[idx];
            completed++;
        }
    }

    // OUTPUT
    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");
    for (i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
               i + 1, at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nGantt Chart:\n");
    for (i = 0; i < k; i++) {
        if (gp[i] == -1) printf("|  -  ");
        else printf("| P%-2d ", gp[i]);
    }
    printf("|\n");

    printf("%d", gt[0]);
    for (i = 1; i <= k; i++) {
        printf("     %d", gt[i]);
    }

    printf("\nAverage WT = %.2f", totalWT / n);
    printf("\nAverage TAT = %.2f\n", totalTAT / n);

    return 0;
}
